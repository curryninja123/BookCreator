%%%%% THEOREM ENVIRONMENTS %%%%%

%%% HELPER ENVIRONMENT FOR THEOREMS
\newenvironment{thmtext}[1]
	{\color{black}
		\transparent{1.0}
			\color{black}
			\scshape
			\begin{thm}[#1]
			\normalfont
	}
	{
		\end{thm}
			
	}
	
%%% MAIN THEOREM ENVIRONMENT
\newenvironment{theorem}[1]
	{\color{primary}
		\transparent{0.3}
		\begin{shaded}
		\begin{leftbar}
			\transparent{1.0}
		\begin{thmtext}{#1}
	}
	{
		\end{thmtext}
		\end{leftbar}
		\end{shaded}
		\transparent{1.0}
	}

%%% HELPER ENVIRONMENT FOR DEFINITIONS
\newenvironment{dfntext}[1]
{\color{black}
	
	\color{black}
	\begin{dfn}[#1]
	}
	{
	\end{dfn}
	
}

%%% MAIN DEFINITION ENVIRONMENT
\newenvironment{definition}[1]
	{\color{primary}
		\transparent{0.3}
		\begin{shaded}
			\begin{leftbar}
				\transparent{1.0}
				\renewcommand{\normalfont}{\scshape}
				\begin{dfntext}{#1}
	}
	{
				\end{dfntext}
			\end{leftbar}
		\end{shaded}
		\transparent{1.0}
	}
